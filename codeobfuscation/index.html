// Deobfuscate HTML code, restoring original JavaScript and CSS
function deobfuscateHtml(code, opts) {
    if (!obfuscationMap.html) throw new Error('No mapping found');
    let html = code;

    let scriptIndex = 0;
    html = html.replace(/)<[^<]*)*<\/script>/gi, script => {
        const jsCode = script.replace(/]*>|<\/script>/g, '');
        if (!jsCode.trim()) return script;
        const mapEntry = obfuscationMap.html.scripts[scriptIndex] || {};
        if (!mapEntry.obfuscated || mapEntry.obfuscated !== jsCode) {
            const deobfuscatedJs = deobfuscateJs(jsCode, opts);
            scriptIndex++;
            return ``;
        }
        scriptIndex++;
        return ``;
    });

    let styleIndex = 0;
    html = html.replace(/)<[^<]*)*<\/style>/gi, style => {
        const cssCode = style.replace(/]*>|<\/style>/g, '');
        if (!cssCode.trim()) return style;
        const mapEntry = obfuscationMap.html.styles[styleIndex] || {};
        if (!mapEntry.obfuscated || mapEntry.obfuscated !== cssCode) {
            const deobfuscatedCss = deobfuscateCss(cssCode, opts);
            styleIndex++;
            return ``;
        }
        styleIndex++;
        return ``;
    });

    if (opts.varRename && obfuscationMap.html.vars) {
        for (const [newVal, val] of Object.entries(obfuscationMap.html.vars)) {
            const regex = new RegExp(`\\b(id|class)=["']([^"']*\\b${newVal}\\b[^"']*)["']`, 'g');
            html = html.replace(regex, (m, attr, value) => {
                const values = value.split(/\s+/);
                const originalValues = values.map(v => v === newVal ? val : v);
                return `${attr}="${originalValues.join(' ')}"`;
            });
        }
    }

    if (opts.strEncrypt && obfuscationMap.html.strings) {
        for (const [encoded, original] of Object.entries(obfuscationMap.html.strings)) {
            const escapedEncoded = encoded.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
            html = html.replace(new RegExp(`>${escapedEncoded}<`, 'g'), `>${original}<`);
        }
    }

    if (opts.deadCode) {
        html = html.replace(/
